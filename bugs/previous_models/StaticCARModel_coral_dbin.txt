
    
    model{
    
    ## priors 
    # CAR prior distribution for spatial random effects:
    rho[1:nsite] ~ car.normal(adj[], weights[], num[], tau)
    
    # hyperprior of rho 
    vrho ~ dnorm(0, 0.5) I(0,)
    tau  <- 1/vrho
    
    # regression parameters
    b0 ~ dunif (0,1)#dnorm(0,1.0E-2)
    
    ## likelihood
    for(i in 1:nsite) {
    
       ## cover dataset 
       C[i] ~ dbin(p[i],N[i])
       logit(p[i])<-b0 + rho[i]
    
    }
    
    
    }
